function analyze(min=0) {
  let a = [];
  for (let evt of EVENTS) {
    let cor = phi(tableFor(evt));
    if (Math.abs(cor) > min)
      a.push(evt +": "+cor.toFixed(4))
  }
  return a
}
function journalEvents() {
  EVENTS.clear();
  for (let entry of JOURNAL)
    for (let e of entry.events)
      EVENTS.add(e);
}
