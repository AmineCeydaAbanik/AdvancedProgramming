function poisson(x, λ, f=4) {
    let p = Math.exp(-λ)
    for (let k=1; k<=x; k++) p = p*λ/k
    return p.toFixed(f)
}
function doTable() {
    let head = a => '      '+ a.join('   ')
    const L = [0.5, 1, 1.5, 2, 2.5, 3]
    let b = [head(L.map((λ) => 'λ='+λ))]
    for (let x=0; x<=8; x++) { 
       let a = L.map(λ => poisson(x, λ))
       b.push('x='+x+'  '+ a.join(' '))
    }
    return b.join('\n')
}
